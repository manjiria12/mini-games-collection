<script>
const EMOJIS=['рЯРґ','рЯР±','рЯР≠','рЯРє','рЯР∞','рЯ¶К','рЯРї','рЯРЉ','рЯР®','рЯРѓ','рЯ¶Б','рЯРЃ','рЯРЈ','рЯРЄ','рЯРµ','рЯРФ','рЯРІ','рЯР¶','рЯ¶Ж','рЯ¶Е','рЯ¶Й','рЯ¶З','рЯРЇ','рЯРЧ','рЯРі','рЯ¶Д','рЯРЭ','рЯРЫ','рЯ¶Л','рЯРМ','рЯРЮ','рЯРЬ','рЯ¶Я','рЯ¶Ч','рЯХЈ','рЯ¶В','рЯРҐ','рЯРН','рЯ¶О','рЯ¶Ц','рЯ¶Х','рЯРЩ','рЯ¶С','рЯ¶Р','рЯ¶Ю','рЯ¶А','рЯР°','рЯР†','рЯРЯ','рЯРђ','рЯР≥','рЯРЛ','рЯ¶И','рЯРК','рЯРЕ','рЯРЖ','рЯ¶У','рЯ¶Н','рЯ¶І','рЯ¶£'];

let pairs=8,cols=4,cards=[],flipped=[],matched=0,moves=0,lock=false,timerInt,seconds=0,started=false,best={8:null,18:null,32:null},peeking=false;

function setDiff(btn){
  document.querySelectorAll('.d-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  pairs=parseInt(btn.dataset.pairs);cols=parseInt(btn.dataset.cols);
  newGame();
}

function newGame(){
  clearInterval(timerInt);seconds=0;moves=0;matched=0;started=false;flipped=[];lock=false;
  updateHUD();
  const emojis=EMOJIS.slice().sort(()=>Math.random()-.5).slice(0,pairs);
  cards=[...emojis,...emojis].sort(()=>Math.random()-.5);
  const grid=document.getElementById('grid');
  grid.className=`grid grid-${cols}`;
  grid.innerHTML='';
  cards.forEach((emoji,i)=>{
    const card=document.createElement('div');
    card.className='card';
    card.dataset.idx=i;
    card.dataset.emoji=emoji;
    card.innerHTML=`<div class="card-inner"><div class="card-back"></div><div class="card-front">${emoji}</div></div>`;
    card.addEventListener('click',()=>flipCard(card));
    grid.appendChild(card);
  });
  document.getElementById('best-moves').textContent=best[pairs]||'вАФ';
}

function flipCard(card){
  if(lock||card.classList.contains('flipped')||card.classList.contains('matched'))return;
  if(!started){started=true;timerInt=setInterval(()=>{seconds++;updateTimer();},1000);}
  card.classList.add('flipped');
  flipped.push(card);
  if(flipped.length===2){
    moves++;
    document.getElementById('moves').textContent=moves;
    lock=true;
    const [a,b]=flipped;
    if(a.dataset.emoji===b.dataset.emoji){
      setTimeout(()=>{a.classList.add('matched');b.classList.add('matched');matched++;document.getElementById('matched').textContent=matched;flipped=[];lock=false;if(matched===pairs)win();},400);
    } else {
      a.classList.add('wrong');b.classList.add('wrong');
      setTimeout(()=>{a.classList.remove('flipped','wrong');b.classList.remove('flipped','wrong');flipped=[];lock=false;},900);
    }
  }
}

function win(){
  clearInterval(timerInt);
  if(!best[pairs]||moves<best[pairs])best[pairs]=moves;
  const div=document.createElement('div');
  div.className='win-overlay';
  const mins=Math.floor(seconds/60),secs=seconds%60;
  div.innerHTML=`
    <div class="win-emoji">рЯОЙ</div>
    <div class="win-title">You Won!</div>
    <div class="win-sub">All ${pairs} pairs matched!</div>
    <div class="win-stats">
      <div class="win-stat"><div class="win-stat-val">${moves}</div><div class="win-stat-lbl">Moves</div></div>
      <div class="win-stat"><div class="win-stat-val">${mins}:${String(secs).padStart(2,'0')}</div><div class="win-stat-lbl">Time</div></div>
      <div class="win-stat"><div class="win-stat-val">${best[pairs]}</div><div class="win-stat-lbl">Best</div></div>
    </div>
    <div class="btn-row">
      <button class="btn btn-primary" onclick="this.closest('.win-overlay').remove();newGame()">Play Again</button>
    </div>`;
  document.body.appendChild(div);
}

function revealAll(){
  if(peeking)return;peeking=true;
  const unmatched=document.querySelectorAll('.card:not(.matched):not(.flipped)');
  unmatched.forEach(c=>c.classList.add('flipped'));
  setTimeout(()=>{unmatched.forEach(c=>c.classList.remove('flipped'));peeking=false;},1200);
}

function updateHUD(){
  document.getElementById('moves').textContent=moves;
  document.getElementById('matched').textContent=matched;
  updateTimer();
}
function updateTimer(){
  const m=Math.floor(seconds/60),s=seconds%60;
  document.getElementById('timer').textContent=`${m}:${String(s).padStart(2,'0')}`;
}
newGame();
