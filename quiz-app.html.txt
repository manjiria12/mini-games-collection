let selectedCat='science',difficulty='easy',questions=[],qIdx=0,score=0,correct=0,timerInt,timeLeft,totalTime=0,answered=false,timings=[];

// Build category cards
const catGrid=document.getElementById('cat-grid');
Object.entries(DATA).forEach(([key,cat])=>{
  const el=document.createElement('div');
  el.className='cat-card'+(key==='science'?' selected':'');
  el.dataset.key=key;
  el.innerHTML=`<div class="cat-icon">${cat.icon}</div><div class="cat-name">${cat.name}</div><div class="cat-count">${cat.questions.length} questions</div>`;
  el.onclick=()=>{document.querySelectorAll('.cat-card').forEach(c=>c.classList.remove('selected'));el.classList.add('selected');selectedCat=key;};
  catGrid.appendChild(el);
});

function setDiff(btn){
  document.querySelectorAll('.diff-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');difficulty=btn.dataset.d;
}

function startQuiz(){
  const pool=DATA[selectedCat].questions.filter(q=>q.d===difficulty);
  if(!pool.length){alert('No questions for this difficulty! Try another.');return;}
  questions=pool.sort(()=>Math.random()-.5).slice(0,Math.min(10,pool.length));
  qIdx=0;score=0;correct=0;totalTime=0;timings=[];
  document.getElementById('home').style.display='none';
  document.getElementById('quiz').style.display='block';
  document.getElementById('results').style.display='none';
  showQ();
}

function showQ(){
  if(qIdx>=questions.length){showResults();return;}
  const q=questions[qIdx];
  answered=false;
  document.getElementById('q-counter').textContent=`Q ${qIdx+1} / ${questions.length}`;
  document.getElementById('progress').style.width=`${(qIdx/questions.length)*100}%`;
  document.getElementById('q-text').textContent=q.q;
  document.getElementById('feedback').textContent='';
  document.getElementById('feedback').className='feedback';
  document.getElementById('next-btn').style.display='none';
  const opts=document.getElementById('options');
  opts.innerHTML='';
  q.o.forEach((opt,i)=>{
    const btn=document.createElement('button');
    btn.className='opt';btn.textContent=opt;
    btn.onclick=()=>answer(i);
    opts.appendChild(btn);
  });
  const tl=difficulty==='easy'?20:difficulty==='medium'?15:10;
  startTimer(tl);
}

function startTimer(sec){
  clearInterval(timerInt);timeLeft=sec;
  const el=document.getElementById('timer');
  el.textContent=sec;el.className='timer';
  timerInt=setInterval(()=>{
    timeLeft--;el.textContent=timeLeft;
    if(timeLeft<=5)el.className='timer warn';
    if(timeLeft<=0){clearInterval(timerInt);timeOut();}
  },1000);
}

function timeOut(){
  if(answered)return;answered=true;
  const q=questions[qIdx];
  timings.push(parseInt(document.getElementById('timer').textContent)||0);
  document.querySelectorAll('.opt').forEach((b,i)=>{b.disabled=true;if(i===q.a)b.classList.add('correct');});
  const fb=document.getElementById('feedback');
  fb.textContent='⏱ Time\'s up! '+q.f;fb.className='feedback show-wrong';
  document.getElementById('next-btn').style.display='block';
}

function answer(idx){
  if(answered)return;answered=true;
  clearInterval(timerInt);
  const elapsed=(difficulty==='easy'?20:difficulty==='medium'?15:10)-timeLeft;
  timings.push(elapsed);
  totalTime+=elapsed;
  const q=questions[qIdx];
  document.querySelectorAll('.opt').forEach((b,i)=>{
    b.disabled=true;
    if(i===q.a)b.classList.add('correct');
    else if(i===idx&&idx!==q.a)b.classList.add('wrong');
  });
  const fb=document.getElementById('feedback');
  if(idx===q.a){
    correct++;
    const bonus=Math.max(0,timeLeft);
    score+=10+bonus;
    fb.textContent='✓ Correct! (+' +(10+bonus)+' pts) '+q.f;
    fb.className='feedback show-correct';
  } else {
    fb.textContent='✗ Incorrect. '+q.f;
    fb.className='feedback show-wrong';
  }
  document.getElementById('next-btn').style.display='block';
}

function nextQ(){qIdx++;showQ();}

function showResults(){
  clearInterval(timerInt);
  document.getElementById('quiz').style.display='none';
  document.getElementById('results').style.display='block';
  const pct=Math.round((correct/questions.length)*100);
  const rc=document.getElementById('r-circle');
  rc.querySelector('.result-pct').textContent=pct+'%';
  const titles=['😅 Keep Practicing!','🙂 Getting There!','😊 Good Job!','😎 Great Work!','🏆 Perfect Score!'];
  const msgs=['Don\'t give up, every expert was once a beginner!','You\'re improving! Review and try again.','Solid performance! A bit more study and you\'ll ace it.','Impressive! You really know your stuff.','Flawless! You\'re a true master of this topic.'];
  const tier=pct===100?4:pct>=80?3:pct>=60?2:pct>=40?1:0;
  const colors=['#e8472a','#f59e0b','#3b82f6','#1a9e5c','#7c3aed'];
  rc.style.borderColor=colors[tier];
  rc.querySelector('.result-pct').style.color=colors[tier];
  document.getElementById('r-title').textContent=titles[tier];
  document.getElementById('r-msg').textContent=msgs[tier];
  document.getElementById('r-correct').textContent=`${correct}/${questions.length}`;
  document.getElementById('r-time').textContent=timings.length?Math.round(timings.reduce((a,b)=>a+b,0)/timings.length)+'s':'—';
  document.getElementById('r-pts').textContent=score;
}

function replay(){
  document.getElementById('results').style.display='none';
  document.getElementById('quiz').style.display='block';
  startQuiz();
}
function goHome(){
  document.getElementById('results').style.display='none';
  document.getElementById('home').style.display='block';
}
