import sys
import subprocess
from pathlib import Path

import pygame

pygame.init()

WIDTH, HEIGHT = 800, 600
SCREEN = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Mini Games Collection")
CLOCK = pygame.time.Clock()

TITLE_FONT = pygame.font.SysFont("arial", 48, bold=True)
SUBTITLE_FONT = pygame.font.SysFont("arial", 24)
BUTTON_FONT = pygame.font.SysFont("arial", 30, bold=True)
SMALL_FONT = pygame.font.SysFont("arial", 20)

BG = (20, 22, 30)
PANEL = (32, 36, 48)
BUTTON = (70, 130, 180)
BUTTON_HOVER = (100, 160, 210)
TEXT = (245, 245, 245)
MUTED = (180, 185, 200)
ACCENT = (120, 220, 160)
EXIT_COLOR = (200, 80, 80)
EXIT_HOVER = (225, 105, 105)

BASE_DIR = Path(__file__).resolve().parent


class Button:
    def __init__(self, x, y, w, h, text, file_name=None, is_exit=False):
        self.rect = pygame.Rect(x, y, w, h)
        self.text = text
        self.file_name = file_name
        self.is_exit = is_exit

    def draw(self, surface, mouse_pos):
        hovered = self.rect.collidepoint(mouse_pos)

        if self.is_exit:
            color = EXIT_HOVER if hovered else EXIT_COLOR
        else:
            color = BUTTON_HOVER if hovered else BUTTON

        pygame.draw.rect(surface, color, self.rect, border_radius=14)
        pygame.draw.rect(surface, (255, 255, 255), self.rect, width=2, border_radius=14)

        label = BUTTON_FONT.render(self.text, True, TEXT)
        label_rect = label.get_rect(center=self.rect.center)
        surface.blit(label, label_rect)

    def clicked(self, mouse_pos):
        return self.rect.collidepoint(mouse_pos)


def launch_game(file_name: str) -> str | None:
    game_path = BASE_DIR / file_name

    if not game_path.exists():
        return f"Missing file: {file_name}"

    try:
        # Close launcher window while the selected game runs
        pygame.display.quit()

        subprocess.run([sys.executable, str(game_path)], check=False)

        # Re-open launcher window after game closes
        pygame.display.init()
        pygame.display.set_mode((WIDTH, HEIGHT))
        pygame.display.set_caption("Mini Games Collection")
        return None

    except Exception as exc:
        pygame.display.init()
        pygame.display.set_mode((WIDTH, HEIGHT))
        pygame.display.set_caption("Mini Games Collection")
        return f"Could not launch {file_name}: {exc}"


def draw_background():
    SCREEN.fill(BG)

    # Decorative panels
    pygame.draw.rect(SCREEN, PANEL, (80, 60, 640, 480), border_radius=24)
    pygame.draw.rect(SCREEN, ACCENT, (80, 60, 640, 480), width=3, border_radius=24)


def draw_text_center(text, font, color, y):
    label = font.render(text, True, color)
    rect = label.get_rect(center=(WIDTH // 2, y))
    SCREEN.blit(label, rect)


def main():
    buttons = [
        Button(250, 180, 300, 55, "Snake", "snake_game.py"),
        Button(250, 255, 300, 55, "Quiz", "quiz_app.py"),
        Button(250, 330, 300, 55, "Memory", "memory_game.py"),
        Button(250, 405, 300, 55, "Tic Tac Toe", "tic_tac_toe.py"),
        Button(250, 480, 300, 50, "Exit", is_exit=True),
    ]

    status_message = "Select a game to start"

    running = True
    while running:
        mouse_pos = pygame.mouse.get_pos()

        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False

            elif event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
                for button in buttons:
                    if button.clicked(mouse_pos):
                        if button.is_exit:
                            running = False
                        else:
                            error = launch_game(button.file_name)
                            status_message = error or f"Opened {button.text}"

            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_ESCAPE:
                    running = False

        draw_background()
        draw_text_center("Mini Games Collection", TITLE_FONT, TEXT, 110)
        draw_text_center("Python + Pygame Launcher", SUBTITLE_FONT, MUTED, 150)

        for button in buttons:
            button.draw(SCREEN, mouse_pos)

        draw_text_center(status_message, SMALL_FONT, ACCENT if "Missing" not in status_message and "Could not" not in status_message else EXIT_COLOR, 560)

        pygame.display.flip()
        CLOCK.tick(60)

    pygame.quit()
    sys.exit()


if __name__ == "__main__":
    main()
