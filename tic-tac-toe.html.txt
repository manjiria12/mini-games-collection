<script>
const WINS=[[0,1,2],[3,4,5],[6,7,8],[0,3,6],[1,4,7],[2,5,8],[0,4,8],[2,4,6]];
let board=Array(9).fill(null),cur='X',mode='ai',aiDiff='easy',scores={X:0,O:0,D:0},gameOver=false,aiThinking=false;

function setMode(m,btn){
  mode=m;document.querySelectorAll('.mode-btn').forEach(b=>b.classList.remove('active'));btn.classList.add('active');
  document.getElementById('ai-row').style.display=m==='ai'?'flex':'none';
  document.getElementById('name-o').textContent=m==='ai'?'AI':'P2';
  resetAll();
}
function setAIDiff(btn){aiDiff=btn.dataset.d;document.querySelectorAll('.ai-diff').forEach(b=>b.classList.remove('active'));btn.classList.add('active');resetBoard();}

function buildBoard(){
  const b=document.getElementById('board');b.innerHTML='';
  board.forEach((v,i)=>{
    const cell=document.createElement('div');
    cell.className='cell'+(v?' taken'+' '+(v==='X'?'x-cell':'o-cell'):'');
    cell.dataset.i=i;
    if(v)cell.innerHTML=`<span class="cell-inner">${v}</span>`;
    cell.addEventListener('click',()=>play(i));
    b.appendChild(cell);
  });
}
function updateStatus(){
  const el=document.getElementById('status');
  if(gameOver)return;
  if(mode==='ai'){el.className=cur==='X'?'status x-turn':'status o-turn';el.textContent=cur==='X'?'Your turn (X)':'AI is thinking...';}
  else{el.className=cur==='X'?'status x-turn':'status o-turn';el.textContent=cur==='X'?'Player X\'s turn':'Player O\'s turn';}
}
function updateScoreUI(){
  document.getElementById('val-x').textContent=scores.X;
  document.getElementById('val-o').textContent=scores.O;
  document.getElementById('draws').textContent=scores.D;
  document.getElementById('score-x').className='score-box'+(cur==='X'&&!gameOver?' active-player':'');
  document.getElementById('score-o').className='score-box'+(cur==='O'&&!gameOver?' active-player':'');
}

function play(i){
  if(board[i]||gameOver||aiThinking)return;
  if(mode==='ai'&&cur==='O')return;
  board[i]=cur;buildBoard();
  const win=checkWin();
  if(win){endGame(win);return;}
  if(board.every(v=>v)){endGame(null);return;}
  cur=cur==='X'?'O':'X';updateStatus();updateScoreUI();
  if(mode==='ai'&&cur==='O'&&!gameOver){
    aiThinking=true;setTimeout(aiMove,400);
  }
}

function aiMove(){
  aiThinking=false;
  const move=getAIMove();
  board[move]='O';buildBoard();
  const win=checkWin();
  if(win){endGame(win);return;}
  if(board.every(v=>v)){endGame(null);return;}
  cur='X';updateStatus();updateScoreUI();
}

function getAIMove(){
  if(aiDiff==='easy'){
    const empty=board.map((v,i)=>v?null:i).filter(v=>v!==null);
    return empty[Math.floor(Math.random()*empty.length)];
  }
  if(aiDiff==='medium'&&Math.random()<0.3){
    const empty=board.map((v,i)=>v?null:i).filter(v=>v!==null);
    return empty[Math.floor(Math.random()*empty.length)];
  }
  return minimax(board,'O').idx;
}

function minimax(b,player,depth=0,alpha=-Infinity,beta=Infinity){
  const win=checkWinOn(b);
  if(win==='O')return{score:10-depth};
  if(win==='X')return{score:depth-10};
  const empty=b.map((v,i)=>v?null:i).filter(v=>v!==null);
  if(!empty.length)return{score:0};
  let best=player==='O'?{score:-Infinity,idx:-1}:{score:Infinity,idx:-1};
  for(const i of empty){
    const nb=[...b];nb[i]=player;
    const s=minimax(nb,player==='O'?'X':'O',depth+1,alpha,beta);
    s.idx=i;
    if(player==='O'){if(s.score>best.score)best=s;alpha=Math.max(alpha,s.score);}
    else{if(s.score<best.score)best=s;beta=Math.min(beta,s.score);}
    if(beta<=alpha)break;
  }
  return best;
}

function checkWinOn(b){
  for(const [a,c,e] of WINS){if(b[a]&&b[a]===b[c]&&b[a]===b[e])return b[a];}
  return null;
}
function checkWin(){
  for(const combo of WINS){
    const [a,c,e]=combo;
    if(board[a]&&board[a]===board[c]&&board[a]===board[e])return combo;
  }
  return null;
}

function endGame(winCombo){
  gameOver=true;
  const el=document.getElementById('status');
  if(winCombo){
    const winner=board[winCombo[0]];
    scores[winner]++;
    winCombo.forEach(i=>{
      const cells=document.querySelectorAll('.cell');
      cells[i].classList.add('win-cell');
      cells[i].innerHTML=`<span class="cell-inner">${winner}</span>`;
    });
    const whoWon=mode==='ai'?(winner==='X'?'You win! 🎉':'AI wins! 🤖'):`Player ${winner} wins! 🎉`;
    el.textContent=whoWon;el.className='status win';
  } else {
    scores.D++;el.textContent='Draw! 🤝';el.className='status draw';
  }
  updateScoreUI();
  setTimeout(()=>{if(gameOver)resetBoard();},2500);
}

function resetBoard(){
  board=Array(9).fill(null);gameOver=false;aiThinking=false;
  cur='X';buildBoard();updateStatus();updateScoreUI();
}
function resetAll(){scores={X:0,O:0,D:0};resetBoard();}
resetBoard();
